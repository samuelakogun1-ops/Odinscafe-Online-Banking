            const btn = document.getElementById('submitBtn');
            btn.disabled = true;
            btn.innerHTML = 'ENCRYPTING & ROUTING...';

                // Save to Database
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'transfers'), {
                    recipient: document.getElementById('recipientName').value,
                    iban: document.getElementById('iban').value,
                    amount: parseFloat(document.getElementById('amount').value),

                    timestamp: serverTimestamp()

                btn.classList.replace('bg-amber-500', 'bg-green-500');
                
                setTimeout(() => {
                    window.location.href = 'dashboard.html';
                }, 2000);
            } catch (error) {
                console.error(error);
                btn.innerHTML = 'ERROR: NODE TIMEOUT';
                btn.disabled = false;
            }
